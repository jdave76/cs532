import requests
from urllib.parse import urljoin
import random

url = "https://norfolk.gov"  # Replace with target URL
base_url = url

response = requests.get(url)
soup = BeautifulSoup(response.content, 'html.parser')

target_unique = 500
unique_uris = set()
uri_list = []

# Extract all links
links = soup.find_all('a', href=True)
uri_list = []


while len(uri_list)<500: 
    for link in links:  
        try: 
            head_response = requests.head(url, allow_redirects=True, timeout=20)
            content_type = head_response.headers.get('Content-Type', '')
            content_length = int(head_response.headers.get('Content-Length', 0)) 
    # Check if it's HTML (case-insensitive)
            if'text/html' in content_type.lower() and content_length>1000:
                uri_list.append(link)
                seed_value = hash(tuple(uri_list))  # Convert list to tuple (hashable) and hash it
                random.seed(seed_value)
        except FileNotFoundError:
            print("File not found.")
